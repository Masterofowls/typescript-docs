linkAdvanced Types

TypeScript's type system goes far beyond simple annotations. These advanced features let you model complex domains with precision.

linkTemplate Literal Types

Construct new string types from existing ones:

1linktype EventName = "click" | "focus" | "blur";

2linktype Handler = `on${Capitalize<EventName>}`; // "onClick" | "onFocus" | "onBlur"

3link

4link// CSS property types

5linktype CSSProperty = "margin" | "padding" | "border";

6linktype CSSDirection = "top" | "right" | "bottom" | "left";

7linktype CSSLonghand = `${CSSProperty}-${CSSDirection}`;

8link// "margin-top" | "margin-right" | ... | "border-left"

9link

10link// URL builder

11linktype Endpoint = `/api/${"users" | "products" | "orders"}`;

12link// "/api/users" | "/api/products" | "/api/orders"

13link

14link// Route parameters

15linktype RouteParams<T extends string> =

16link  T extends `${infer _Start}:${infer Param}/${infer Rest}`

17link    ? { [K in Param | keyof RouteParams<Rest>]: string }

18link    : T extends `${infer _Start}:${infer Param}`

19link    ? { [K in Param]: string }

20link    : {};

21link

22linktype Params = RouteParams<"/users/:id/posts/:postId">;

23link// { id: string; postId: string }

linkDiscriminated Unions

A pattern for type-safe variant types:

1link// Each variant has a unique discriminant field

2linktype Shape =

3link  | { kind: "circle";    radius: number }

4link  | { kind: "square";    side: number }

5link  | { kind: "rectangle"; width: number; height: number };

6link

7linkfunction area(shape: Shape): number {

8link  switch (shape.kind) {

9link    case "circle":    return Math.PI * shape.radius ** 2;

10link    case "square":    return shape.side ** 2;

11link    case "rectangle": return shape.width * shape.height;

12link    // TypeScript knows all cases are handled — no default needed

13link  }

14link}

15link

16link// Result type (like Rust's Result<T, E>)

17linktype Result<T, E = Error> =

18link  | { success: true;  value: T }

19link  | { success: false; error: E };

20link

21linkfunction parseJSON<T>(text: string): Result<T> {

22link  try {

23link    return { success: true, value: JSON.parse(text) };

24link  } catch (e) {

25link    return { success: false, error: e as Error };

26link  }

27link}

28link

29linkconst result = parseJSON<{ name: string }>('{"name":"Alice"}');

30linkif (result.success) {

31link  console.log(result.value.name); // TypeScript knows .value exists

32link} else {

33link  console.error(result.error.message); // TypeScript knows .error exists

34link}

linkBranded / Nominal Types

Prevent mixing up values with the same underlying type:

1link// Brand helper

2linktype Brand<T, B extends string> = T & { readonly __brand: B };

3link

4linktype UserId    = Brand<string, "UserId">;

5linktype OrderId   = Brand<string, "OrderId">;

6linktype ProductId = Brand<string, "ProductId">;

7linktype Email     = Brand<string, "Email">;

8link

9link// Constructor functions

10linkfunction createUserId(id: string): UserId {

11link  return id as UserId;

12link}

13linkfunction createEmail(email: string): Email {

14link  if (!email.includes("@")) throw new Error("Invalid email");

15link  return email as Email;

16link}

17link

18linkfunction getUser(id: UserId): void { /* ... */ }

19linkfunction getOrder(id: OrderId): void { /* ... */ }

20link

21linkconst userId = createUserId("user-123");

22linkconst orderId = "order-456" as OrderId;

23link

24linkgetUser(userId);    // OK

25link// getUser(orderId); // Error: OrderId ≠ UserId

linkRecursive Types

1link// Recursive tree

2linktype TreeNode<T> = {

3link  value: T;

4link  children: TreeNode<T>[];

5link};

6link

7link// Recursive readonly

8linktype DeepReadonly<T> = {

9link  readonly [K in keyof T]: T[K] extends object ? DeepReadonly<T[K]> : T[K];

10link};

11link

12link// Deep partial

13linktype DeepPartial<T> = {

14link  [K in keyof T]?: T[K] extends object ? DeepPartial<T[K]> : T[K];

15link};

16link

17link// Flatten nested object to dot-notation keys

18linktype DotNotation<T, Prefix extends string = ""> = {

19link  [K in keyof T & string]: T[K] extends object

20link    ? DotNotation<T[K], `${Prefix}${K}.`>

21link    : `${Prefix}${K}`;

22link}[keyof T & string];

23link

24linkinterface Config {

25link  server: { host: string; port: number };

26link  db: { uri: string };

27link}

28link

29linktype ConfigKeys = DotNotation<Config>;

30link// "server.host" | "server.port" | "db.uri"

linkType Guards

Narrow types at runtime:

1link// typeof guard

2linkfunction processInput(value: string | number | boolean) {

3link  if (typeof value === "string") {

4link    return value.toUpperCase(); // string

5link  } else if (typeof value === "number") {

6link    return value.toFixed(2);    // number

7link  }

8link  return String(value);         // boolean

9link}

10link

11link// instanceof guard

12linkclass HttpError extends Error {

13link  constructor(public statusCode: number, message: string) {

14link    super(message);

15link  }

16link}

17link

18linkfunction handleError(error: unknown) {

19link  if (error instanceof HttpError) {

20link    console.log(`HTTP ${error.statusCode}: ${error.message}`);

21link  } else if (error instanceof Error) {

22link    console.log(`Error: ${error.message}`);

23link  } else {

24link    console.log("Unknown error:", error);

25link  }

26link}

27link

28link// Custom type guard (predicate)

29linkinterface Cat { meow(): void }

30linkinterface Dog { bark(): void }

31link

32linkfunction isCat(animal: Cat | Dog): animal is Cat {

33link  return "meow" in animal;

34link}

35link

36linkfunction makeNoise(animal: Cat | Dog) {

37link  if (isCat(animal)) {

38link    animal.meow(); // TypeScript knows it's Cat

39link  } else {

40link    animal.bark(); // TypeScript knows it's Dog

41link  }

42link}

43link

44link// Assertion function

45linkfunction assertDefined<T>(value: T | null | undefined, message: string): asserts value is T {

46link  if (value === null || value === undefined) {

47link    throw new Error(message);

48link  }

49link}

linkSatisfies Operator (TypeScript 4.9+)

1linktype Color = "red" | "green" | "blue";

2link

3link// 'satisfies' validates the type without widening

4linkconst palette = {

5link  red:   [255, 0, 0],

6link  green: "#00ff00",

7link  blue:  [0, 0, 255],

8link} satisfies Record<Color, string | number[]>;

9link

10link// palette.red is number[], not string | number[]

11linkconst r = palette.red;   // type: number[]

12linkconst g = palette.green; // type: string
