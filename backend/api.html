3linkimport cors from "cors";

5linkimport compression from "compression";

7linkimport routes from "./routes";

8linkimport { errorHandler, notFound } from "./middleware/error.middleware";

9linkimport { requestId, requestLogger } from "./middleware/logging.middleware";

10link

11linkexport function createApp(): Express {

12link  const app = express();

13link

14link  // Security middleware

15link  app.use(helmet());

16link  app.use(cors({ origin: process.env.CORS_ORIGIN ?? "*" }));

17link

18link  // Body parsing

19link  app.use(json({ limit: "10mb" }));

20link  app.use(urlencoded({ extended: true }));

21link

22link  // Compression

23link  app.use(compression());

24link

25link  // Logging & tracing

26link  app.use(requestId);

27link  app.use(requestLogger);

28link

29link  // Health check

30link  app.get("/health", (req, res) => {

31link    res.json({ status: "ok", timestamp: new Date().toISOString() });

32link  });

33link

34link  // API routes

35link  app.use("/api/v1", routes);

36link

37link  // 404 handler

38link  app.use(notFound);

39link

40link  // Global error handler (MUST be last)

41link  app.use(errorHandler);

42link

43link  return app;

44link}

7linkexport class UserController {

10link  getAll = async (

12link    res: Response,

13link    next: NextFunction

14link  ): Promise<void> => {

18link      limit: parseInt(limit, 10),

linkService Layer

1link// src/services/user.service.ts

2linkimport bcrypt from "bcryptjs";

3linkimport { UserRepository } from "../repositories/user.repository";

4linkimport { AppError } from "../middleware/error.middleware";

5linkimport type {

6link  User, CreateUserDto, UpdateUserDto,

7link  PaginatedResult, UserQuery

8link} from "../types";

9link

10linkexport class UserService {

11link  constructor(private userRepo = new UserRepository()) {}

12link

13link  async findAll(query: UserQuery): Promise<PaginatedResult<User>> {

14link    const { page = 1, limit = 20, search, role } = query;

15link    const offset = (page - 1) * limit;

16link

17link    const [users, total] = await this.userRepo.findAndCount({

18link      offset,

19link      limit,

20link      where: { search, role },

21link    });

22link

23link    return {

24link      data: users.map(this.sanitize),

25link      pagination: { page, limit, total, totalPages: Math.ceil(total / limit) },

26link    };

27link  }

28link

29link  async findById(id: string): Promise<User | null> {

30link    const user = await this.userRepo.findOne({ where: { id } });

31link    return user ? this.sanitize(user) : null;

32link  }

33link

34link  async create(dto: CreateUserDto): Promise<User> {

35link    const existing = await this.userRepo.findOne({ where: { email: dto.email } });

36link    if (existing) throw new AppError(409, "Email already registered");

37link

38link    const hashed = await bcrypt.hash(dto.password, 12);

39link    const user = await this.userRepo.create({ ...dto, password: hashed });

40link    return this.sanitize(user);

41link  }

42link

43link  async update(id: string, dto: UpdateUserDto): Promise<User | null> {

44link    const user = await this.userRepo.findOne({ where: { id } });

45link    if (!user) return null;

46link

47link    if (dto.password) {

48link      dto.password = await bcrypt.hash(dto.password, 12);

49link    }

50link

51link    const updated = await this.userRepo.update(id, dto);

52link    return updated ? this.sanitize(updated) : null;

53link  }

54link

55link  async delete(id: string): Promise<void> {

56link    const user = await this.userRepo.findOne({ where: { id } });

57link    if (!user) throw new AppError(404, "User not found");

58link    await this.userRepo.delete(id);

59link  }

60link

61link  // Remove sensitive fields before returning

62link  private sanitize(user: any): User {

63link    const { password, __v, ...safe } = user;

64link    return safe;

65link  }

66link}

12link

13link// Parse and validate environment — throws on invalid

15link

17link  console.error("❌ Invalid environment variables:");

20link}

23linkexport type Config = z.infer<typeof EnvSchema>;
