linkTyped Request Params, Body, Query

1linkimport { Request, Response } from "express";

2link

3link// Type-safe request shapes

4linkinterface CreateUserBody {

5link  name: string;

6link  email: string;

7link  password: string;

8link  role?: "admin" | "user";

9link}

10link

11linkinterface UserParams {

12link  id: string;

13link}

14link

15linkinterface UserQuery {

16link  page?: string;

17link  limit?: string;

18link  search?: string;

19link}

20link

21link// TypeScript knows exact types for each part of the request

22linkrouter.post(

23link  "/",

24link  async (

25link    req: Request<{}, {}, CreateUserBody>,   // params, response, body

26link    res: Response

27link  ) => {

28link    const { name, email, password, role = "user" } = req.body;

29link

30link    // TypeScript knows: name is string, role is "admin" | "user" | undefined

31link    const user = await UserService.create({ name, email, password, role });

32link    res.status(201).json({ data: user });

33link  }

34link);

35link

36linkrouter.get(

37link  "/",

38link  async (

39link    req: Request<{}, {}, {}, UserQuery>,    // 4th generic = query

40link    res: Response

41link  ) => {

42link    const page = parseInt(req.query.page ?? "1", 10);

43link    const limit = parseInt(req.query.limit ?? "20", 10);

44link    const search = req.query.search;

45link

46link    const result = await UserService.findAll({ page, limit, search });

47link    res.json(result);

48link  }

linkMiddleware

1link// src/middleware/auth.middleware.ts

2linkimport { Request, Response, NextFunction } from "express";

3link

4link// Extend Request with custom fields

5linkdeclare global {

6link  namespace Express {

7link    interface Request {

8link      user?: JwtPayload;

9link      requestId?: string;

10link    }

11link  }

12link}

13link

14linkinterface JwtPayload {

15link  id: string;

16link  email: string;

17link  role: "admin" | "user";

18link}

19link

20linkexport function authenticate(

21link  req: Request,

22link  res: Response,

23link  next: NextFunction

24link): void {

25link  const token = req.headers.authorization?.replace("Bearer ", "");

26link

27link  if (!token) {

28link    res.status(401).json({ error: "No token provided" });

29link    return;

30link  }

31link

32link  try {

33link    const payload = verifyToken(token) as JwtPayload;

34link    req.user = payload;

35link    next();

36link  } catch {

37link    res.status(401).json({ error: "Invalid token" });

38link  }

39link}

40link

41linkexport function authorize(...roles: string[]) {

42link  return (req: Request, res: Response, next: NextFunction): void => {

43link    if (!req.user) {

44link      res.status(401).json({ error: "Unauthenticated" });

45link      return;

46link    }

47link

48link    if (!roles.includes(req.user.role)) {

49link      res.status(403).json({ error: "Insufficient permissions" });

50link      return;

51link    }

52link

53link    next();

54link  };

55link}

56link

57link// Attach unique request ID

58linkexport function requestId(req: Request, res: Response, next: NextFunction): void {

59link  req.requestId = crypto.randomUUID();

60link  res.setHeader("X-Request-ID", req.requestId);

61link  next();

62link}

linkValidation Middleware

1link// src/middleware/validate.middleware.ts

2linkimport { Request, Response, NextFunction } from "express";

3linkimport { z, ZodSchema } from "zod";

4link

5linkexport function validate<T>(schema: ZodSchema<T>) {

6link  return (req: Request, res: Response, next: NextFunction): void => {

7link    const result = schema.safeParse(req.body);

8link

9link    if (!result.success) {

10link      const errors = result.error.errors.map((e) => ({

11link        field: e.path.join("."),

12link        message: e.message,

13link      }));

14link      res.status(400).json({ error: "Validation failed", details: errors });

15link      return;

16link    }

17link

18link    req.body = result.data; // replace with parsed & typed data

19link    next();

20link  };

21link}

22link

23link// Usage with Zod schemas

24linkconst CreateUserSchema = z.object({

25link  name:     z.string().min(2).max(100),

26link  email:    z.string().email(),

27link  password: z.string().min(8),

28link  role:     z.enum(["admin", "user"]).optional().default("user"),

29link});

30link

31linkrouter.post("/users", validate(CreateUserSchema), async (req, res) => {

32link  const user = await UserService.create(req.body);

33link  res.status(201).json({ data: user });

34link});

linkError Handling

1link// src/middleware/error.middleware.ts

2linkimport { Request, Response, NextFunction } from "express";

3link

4linkexport class AppError extends Error {

5link  constructor(

6link    public statusCode: number,

7link    public message: string,

8link    public isOperational = true,

9link  ) {

10link    super(message);

11link    Object.setPrototypeOf(this, AppError.prototype);

12link  }

13link}

14link

15linkexport class NotFoundError extends AppError {

16link  constructor(resource: string) {

17link    super(404, `${resource} not found`);

18link  }

19link}

20link

21linkexport class ValidationError extends AppError {

22link  constructor(message: string) {

23link    super(400, message);

24link  }

25link}

26link

27link// Global error handler — MUST have 4 params

28linkexport function errorHandler(

29link  error: Error | AppError,

30link  req: Request,

31link  res: Response,

32link  next: NextFunction,   // Must include even if unused

33link): void {

34link  if (error instanceof AppError) {

35link    res.status(error.statusCode).json({

36link      error: error.message,

37link      ...(process.env.NODE_ENV === "development" && { stack: error.stack }),

38link    });

39link    return;

40link  }

41link

42link  console.error("Unhandled error:", error);

43link  res.status(500).json({ error: "Internal server error" });

44link}

45link

46link// Async error wrapper — catches async errors and passes to next()

47linkexport function asyncHandler<P, ResBody, ReqBody, ReqQuery>(

48link  fn: (req: Request<P, ResBody, ReqBody, ReqQuery>, res: Response, next: NextFunction) => Promise<void>

49link) {

50link  return (

51link    req: Request<P, ResBody, ReqBody, ReqQuery>,

52link    res: Response,

53link    next: NextFunction

54link  ) => {

55link    fn(req, res, next).catch(next);

56link  };

57link}

linkComplete Router Example

1link// src/routes/user.routes.ts

2linkimport { Router } from "express";

3linkimport { authenticate, authorize } from "../middleware/auth.middleware";

4linkimport { validate } from "../middleware/validate.middleware";

5linkimport { asyncHandler } from "../middleware/error.middleware";

6linkimport { UserController } from "../controllers/user.controller";

7linkimport { CreateUserSchema, UpdateUserSchema } from "../schemas/user.schema";

8link

9linkconst router = Router();

10linkconst controller = new UserController();

11link

12linkrouter.get(    "/",     authenticate, asyncHandler(controller.getAll));

13linkrouter.get(    "/:id",  authenticate, asyncHandler(controller.getOne));

14linkrouter.post(   "/",     authenticate, authorize("admin"), validate(CreateUserSchema), asyncHandler(controller.create));

15linkrouter.put(    "/:id",  authenticate, validate(UpdateUserSchema), asyncHandler(controller.update));

16linkrouter.delete( "/:id",  authenticate, authorize("admin"), asyncHandler(controller.delete));

17link

18linkexport default router;
