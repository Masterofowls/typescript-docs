linkConstructor Shorthand

TypeScript lets you declare and initialise properties directly in the constructor parameter list:

1linkclass Person {

2link  constructor(

3link    public name: string,         // public: readable & writable from outside

4link    private age: number,         // private: only inside this class

5link    protected email: string,     // protected: this class + subclasses

6link    readonly id: string = "abc", // readonly: set once, never again

7link  ) {}

8link

9link  getAge(): number {

10link    return this.age; // OK: inside the class

11link  }

12link}

13link

14linkconst p = new Person("Alice", 30, "a@x.com");

15linkconsole.log(p.name);  // OK

16link// p.age;             // Error: 'age' is private

17link// p.email;           // Error: 'email' is protected

18link// p.id = "xyz";      // Error: 'id' is readonly

3link

4link  get balance(): number {

5link    return this._balance;

6link  }

7link

8link  set balance(amount: number) {

9link    if (amount < 0) {

10link      throw new RangeError("Balance cannot be negative");

11link    }

12link    this._balance = amount;

13link  }

14link

15link  deposit(amount: number): void {

16link    if (amount <= 0) throw new RangeError("Deposit must be positive");

17link    this._balance += amount;

18link  }

19link

20link  withdraw(amount: number): void {

21link    if (amount > this._balance) throw new Error("Insufficient funds");

22link    this._balance -= amount;

23link  }

24link}

25link

26linkconst account = new BankAccount();

27linkaccount.deposit(100);

28linkconsole.log(account.balance); // 100

29linkaccount.balance = 200;        // Uses setter

2link  serialize(): string;

5link

6linkinterface Loggable {

7link  log(): void;

9link

10linkclass User implements Serializable, Loggable {

linkAbstract Classes

1linkabstract class BaseRepository<T extends { id: string }> {

2link  protected items: Map<string, T> = new Map();

3link

4link  abstract validate(item: T): boolean;

5link

6link  save(item: T): T {

7link    if (!this.validate(item)) {

8link      throw new Error("Validation failed");

9link    }

10link    this.items.set(item.id, item);

11link    return item;

12link  }

13link

14link  findById(id: string): T | undefined {

15link    return this.items.get(id);

16link  }

17link

18link  findAll(): T[] {

19link    return Array.from(this.items.values());

20link  }

21link

22link  delete(id: string): boolean {

23link    return this.items.delete(id);

24link  }

25link}

26link

27linkinterface Product {

28link  id: string;

29link  name: string;

30link  price: number;

31link}

32link

33linkclass ProductRepository extends BaseRepository<Product> {

34link  validate(product: Product): boolean {

35link    return product.name.length > 0 && product.price >= 0;

36link  }

37link}

38link

39linkconst repo = new ProductRepository();

40linkrepo.save({ id: "1", name: "Widget", price: 9.99 });

41linkconsole.log(repo.findAll()); // [{ id: "1", name: "Widget", price: 9.99 }]

linkMixins

Combine multiple behaviors without deep inheritance:

1linktype Constructor<T = {}> = new (...args: any[]) => T;

2link

3linkfunction Timestamped<TBase extends Constructor>(Base: TBase) {

4link  return class extends Base {

5link    createdAt = new Date();

6link    updatedAt = new Date();

7link

8link    touch() {

9link      this.updatedAt = new Date();

10link    }

11link  };

12link}

13link

14linkfunction Activatable<TBase extends Constructor>(Base: TBase) {

15link  return class extends Base {

16link    isActive = false;

17link

18link    activate() {

19link      this.isActive = true;

20link    }

21link

22link    deactivate() {

23link      this.isActive = false;

24link    }

25link  };

26link}

27link

28linkclass User {

29link  constructor(public name: string) {}

30link}

31link

32linkconst TimestampedActivatableUser = Activatable(Timestamped(User));

33link

34linkconst user = new TimestampedActivatableUser("Alice");

35linkuser.activate();

36linkconsole.log(user.isActive);   // true

37linkconsole.log(user.createdAt);  // Date object
