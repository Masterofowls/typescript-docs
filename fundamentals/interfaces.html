linkInterfaces

Interfaces define contracts — they describe the shape an object must conform to. Mastering interfaces is essential for writing maintainable TypeScript.

linkBasic Interface

1linkinterface User {

2link  id: number;

3link  name: string;

4link  email: string;

5link}

6link

7link// Object must match the interface exactly

8linkconst alice: User = {

9link  id: 1,

10link  name: "Alice",

11link  email: "alice@example.com",

12link};

linkOptional Properties

1linkinterface Config {

2link  host: string;

3link  port: number;

4link  timeout?: number;   // Optional — may be undefined

5link  retries?: number;

6link}

7link

8linkconst cfg: Config = { host: "localhost", port: 3000 };

9link// timeout and retries are not required

linkReadonly Properties

1linkinterface Point {

2link  readonly x: number;

3link  readonly y: number;

4link}

5link

6linkconst p: Point = { x: 10, y: 20 };

7link// p.x = 5; // Error: Cannot assign to 'x' because it is a read-only property

linkIndex Signatures

When you don't know all property names ahead of time:

1linkinterface StringMap {

2link  [key: string]: string;

3link}

4link

5linkinterface NumberMap {

6link  [key: string]: number;

7link}

8link

9linkconst headers: StringMap = {

10link  "Content-Type": "application/json",

11link  Authorization: "Bearer abc123",

12link};

13link

14link// Mixed: known + dynamic

15linkinterface Settings {

16link  id: number;                  // Known property

17link  [key: string]: unknown;      // Any other key

18link}

linkMethod Signatures

1linkinterface Logger {

2link  log(message: string): void;

3link  warn(message: string): void;

4link  error(message: string, error?: Error): void;

5link}

6link

7link// Method shorthand vs property function

8linkinterface Formatter {

9link  format(value: unknown): string;  // Method signature

10link  transform: (value: string) => string; // Property with function type

11link}

linkExtending Interfaces

Interfaces can extend one or multiple other interfaces:

1linkinterface Animal {

2link  name: string;

3link  sound(): string;

4link}

5link

6linkinterface Pet extends Animal {

7link  owner: string;

8link  vaccinated: boolean;

9link}

10link

11linkinterface ServiceDog extends Pet {

12link  certificationId: string;

13link  task: string;

14link}

15link

16linkconst guide: ServiceDog = {

17link  name: "Rex",

18link  sound: () => "Woof",

19link  owner: "John",

20link  vaccinated: true,

21link  certificationId: "SD-2024-001",

22link  task: "Guide for visually impaired",

23link};

linkInterface Merging (Declaration Merging)

TypeScript merges multiple declarations of the same interface:

1linkinterface Window {

2link  title: string;

3link}

4link

5linkinterface Window {

6link  history: History;

7link}

8link

9link// Merged: Window has BOTH title and history

10linkconst w: Window; // { title: string; history: History }

This is how @types packages extend browser globals.

linkInterface vs Type Alias

1link// Interface — extendable, declaration-mergeable

2linkinterface Animal {

3link  name: string;

4link}

5linkinterface Animal {

6link  legs: number; // OK: merges

7link}

8link

9link// Type alias — more flexible but can't merge

10linktype Vehicle = {

11link  brand: string;

12link};

13link// type Vehicle = { model: string }; // Error: duplicate

14link

15link// Prefer interface for object shapes used as contracts

16link// Prefer type for unions, intersections, and primitives

linkFunctional Interfaces

1link// Call signature

2linkinterface Transformer<T, U> {

3link  (input: T): U;

4link}

5link

6linkconst toNumber: Transformer<string, number> = (s) => parseInt(s, 10);

7link

10link  new (hour: number, minute: number): ClockInterface;

11link}

linkGeneric Interfaces

1linkinterface Repository<T> {

2link  findById(id: string): Promise<T | null>;

3link  findAll(): Promise<T[]>;

4link  create(entity: Omit<T, "id">): Promise<T>;

5link  update(id: string, entity: Partial<T>): Promise<T | null>;

6link  delete(id: string): Promise<boolean>;

7link}

8link

9linkinterface User {

10link  id: string;

11link  name: string;

12link  email: string;

13link}

14link

15linkclass UserRepository implements Repository<User> {

16link  async findById(id: string) { /* ... */ return null; }

17link  async findAll() { return []; }

18link  async create(user: Omit<User, "id">) { return { id: "1", ...user }; }

19link  async update(id: string, user: Partial<User>) { return null; }

20link  async delete(id: string) { return true; }

21link}

linkReal-World Interface Example

1link// API response shapes

2linkinterface ApiResponse<T> {

3link  data: T;

4link  status: number;

5link  message: string;

6link  timestamp: string;

7link}

8link

9linkinterface PaginatedResponse<T> extends ApiResponse<T[]> {

10link  pagination: {

11link    page: number;

12link    limit: number;

13link    total: number;

14link    totalPages: number;

15link  };

16link}

17link

18link// Service contract

19linkinterface UserService {

20link  getUser(id: string): Promise<ApiResponse<User>>;

21link  getUsers(page: number, limit: number): Promise<PaginatedResponse<User>>;

22link  createUser(data: CreateUserDto): Promise<ApiResponse<User>>;

23link  updateUser(id: string, data: UpdateUserDto): Promise<ApiResponse<User>>;

24link  deleteUser(id: string): Promise<ApiResponse<null>>;

25link}

26link

27link// DTOs (Data Transfer Objects)

28linkinterface CreateUserDto {

29link  name: string;

30link  email: string;

31link  password: string;

32link  role?: "admin" | "user";

33link}

34link

35linkinterface UpdateUserDto extends Partial<Omit<CreateUserDto, "password">> {

36link  currentPassword?: string;

37link  newPassword?: string;

38link}
