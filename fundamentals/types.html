linkTypeScript Types

TypeScript's type system is its superpower. Understanding types at every level — from primitives to advanced utility types — is what separates TypeScript beginners from pros.

linkPrimitive Types

1link// The 8 primitive types in TypeScript

2linklet isDone: boolean = false;

3linklet age: number = 25;

4linklet bigNumber: bigint = 9007199254740991n;

5linklet name: string = "Alice";

6linklet nothing: null = null;

7linklet notDefined: undefined = undefined;

8linklet unique: symbol = Symbol("id");

9link

10link// 'object' — any non-primitive value

11linklet obj: object = { a: 1 };

linkType Inference

TypeScript infers types when you don't annotate them:

1linklet count = 0;          // inferred: number

2linklet greeting = "Hello"; // inferred: string

3linklet active = true;      // inferred: boolean

4link

5link// Inference in functions

6linkfunction double(x: number) {

7link  return x * 2; // return type inferred as 'number'

8link}

linkTuples

A tuple is an array with a fixed number of elements and known types:

1linklet pair: [string, number] = ["Alice", 30];

2linklet rgb: [number, number, number] = [255, 128, 0];

3link

4link// Named tuples (TypeScript 4.0+)

5linklet user: [name: string, age: number] = ["Bob", 25];

6linkconsole.log(user[0]); // Bob

7link

8link// Optional tuple elements

9linktype StringNumber = [string, number?];

10linkconst a: StringNumber = ["hello"];

11linkconst b: StringNumber = ["hello", 42];

linkUnion Types

A value can be one of several types:

1linktype ID = string | number;

2link

3linkfunction printId(id: ID): void {

4link  if (typeof id === "string") {

5link    console.log(id.toUpperCase()); // TypeScript knows id is string here

6link  } else {

7link    console.log(id.toFixed(2));    // TypeScript knows id is number here

8link  }

9link}

10link

11link// Union with literals

12linktype Direction = "north" | "south" | "east" | "west";

13linktype Status = "pending" | "active" | "inactive";

14link

15linkfunction move(dir: Direction): void {

16link  console.log(`Moving ${dir}`);

17link}

linkIntersection Types

Combine multiple types into one:

1linktype HasName = { name: string };

2linktype HasAge  = { age: number };

3link

4linktype Person = HasName & HasAge;

5link

6linkconst person: Person = { name: "Alice", age: 30 }; // Must have BOTH

7link

8link// Practical use: Middleware + Request

9linktype AuthRequest = Request & { user: { id: string; role: string } };

linkLiteral Types

A type that accepts only one specific value:

1linktype Yes = "yes";

2linktype One = 1;

3linktype True = true;

4link

5link// Useful in discriminated unions

6linktype Shape =

7link  | { kind: "circle";    radius: number }

8link  | { kind: "rectangle"; width: number; height: number };

9link

10linkfunction area(shape: Shape): number {

11link  switch (shape.kind) {

12link    case "circle":    return Math.PI * shape.radius ** 2;

13link    case "rectangle": return shape.width * shape.height;

14link  }

15link}

linkThe any, unknown, and never Types

1link// any — disables type checking (avoid!)

2linklet anything: any = 42;

3linkanything = "now a string"; // No error

4linkanything.someMethod();     // No error — dangerous!

5link

6link// unknown — type-safe any (use this instead)

7linklet value: unknown = fetchData();

8link// value.toString();           // Error: must narrow first

9linkif (typeof value === "string") {

10link  console.log(value.toUpperCase()); // OK after narrowing

11link}

12link

13link// never — a value that never occurs

14linkfunction throwError(message: string): never {

15link  throw new Error(message);

16link}

17link

18link// Exhaustiveness check with never

19linkfunction assertNever(x: never): never {

20link  throw new Error("Unexpected value: " + x);

21link}

linkUtility Types

TypeScript ships with powerful built-in utility types:

1linkinterface User {

2link  id: number;

3link  name: string;

4link  email: string;

5link  password: string;

6link  role: "admin" | "user";

7link}

8link

9link// Partial — all properties optional

10linktype UpdateUser = Partial<User>;

11link

12link// Required — all properties required

13linktype RequiredUser = Required<User>;

14link

15link// Readonly — prevent mutation

16linktype FrozenUser = Readonly<User>;

17link

18link// Pick — select a subset of properties

19linktype PublicUser = Pick<User, "id" | "name" | "email">;

20link

21link// Omit — exclude properties

22linktype SafeUser = Omit<User, "password">;

23link

24link// Record — key-value map

25linktype RoleMap = Record<string, User[]>;

26link

27link// Exclude — from union

28linktype NonAdmin = Exclude<User["role"], "admin">; // "user"

29link

30link// Extract — from union

31linktype AdminOnly = Extract<User["role"], "admin">; // "admin"

32link

33link// NonNullable — remove null/undefined

34linktype DefinedString = NonNullable<string | null | undefined>; // string

35link

36link// ReturnType — extract return type of function

37linkfunction getUser() { return { id: 1, name: "Alice" }; }

38linktype UserShape = ReturnType<typeof getUser>; // { id: number; name: string }

39link

40link// Parameters — extract parameter types

41linktype GetUserParams = Parameters<typeof getUser>; // []

42link

43link// Awaited — unwrap Promise

44linktype ResolvedUser = Awaited<Promise<User>>; // User
