The entry point of most TypeScript applications is index.ts. Let's walk through creating one from scratch.

linkMinimal index.ts

1link// index.ts

2linkconsole.log("Hello, TypeScript!");

Run it:

1link$ts-node index.ts

2link$# Output: Hello, TypeScript!

linkA Real-World index.ts

In production apps, index.ts typically bootstraps the application — it wires together your modules, starts a server, or runs the main program logic.

1link// index.ts

2linkimport { createApp } from "./app";

3linkimport { connectDatabase } from "./database";

4linkimport { config } from "./config";

5link

6linkasync function main(): Promise<void> {

7link  try {

8link    // 1. Connect to database

9link    await connectDatabase(config.db.uri);

10link    console.log("✅ Database connected");

11link

12link    // 2. Create and configure Express app

13link    const app = createApp();

14link

15link    // 3. Start listening

16link    app.listen(config.port, () => {

17link      console.log(`🚀 Server running on port ${config.port}`);

18link    });

19link  } catch (error) {

20link    console.error("❌ Failed to start application:", error);

21link    process.exit(1);

22link  }

23link}

24link

25linkmain();

linktsconfig.json Essentials

1link{

2link  "compilerOptions": {

3link    "target": "ES2022",           // JavaScript version to compile to

4link    "module": "commonjs",         // Module system

5link    "lib": ["ES2022"],            // Type definitions included

6link    "outDir": "./dist",           // Output directory

7link    "rootDir": "./src",           // Source directory

8link    "strict": true,               // Enable all strict checks

9link    "esModuleInterop": true,      // Allow default imports from CJS modules

10link    "skipLibCheck": true,         // Skip type-checking .d.ts files

11link    "forceConsistentCasingInFileNames": true,

12link    "resolveJsonModule": true,    // Import JSON files

13link    "declaration": true,          // Generate .d.ts files

14link    "sourceMap": true             // Generate source maps for debugging

15link  },

16link  "include": ["src/**/*"],

17link  "exclude": ["node_modules", "dist", "**/*.test.ts"]

18link}

linkStrict Mode Options

1link{

2link  "compilerOptions": {

3link    "strict": true,               // Enables all the below:

4link    "noImplicitAny": true,        // Error on implicit 'any' types

5link    "strictNullChecks": true,     // null/undefined are distinct types

6link    "strictFunctionTypes": true,  // Function parameter variance

7link    "strictBindCallApply": true,  // Type-check bind/call/apply

8link    "noImplicitThis": true,       // Error on 'this' with implicit any

9link    "alwaysStrict": true          // Emit 'use strict' in output

10link  }

11link}
