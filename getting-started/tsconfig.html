2link  "compilerOptions": {

3link    // ─── Output ───────────────────────────────────

4link    "target": "ES2022",          // Compile to this JS version

5link    "module": "commonjs",        // Module format: commonjs | ESNext | AMD

6link    "outDir": "./dist",          // Where to put compiled files

7link    "rootDir": "./src",          // Root of source files

8link    "declaration": true,         // Generate .d.ts type files

9link    "declarationDir": "./types", // Where to put .d.ts files

10link    "sourceMap": true,           // .js.map for debugging

11link

12link    // ─── Type Checking ────────────────────────────

13link    "strict": true,              // All strict checks (recommended)

14link    "noImplicitAny": true,       // Disallow implicit any

15link    "strictNullChecks": true,    // null ≠ undefined ≠ string

16link    "noUnusedLocals": true,      // Error on unused variables

17link    "noUnusedParameters": true,  // Error on unused function params

18link    "noImplicitReturns": true,   // All code paths must return

19link    "noFallthroughCasesInSwitch": true,

20link

21link    // ─── Module Resolution ────────────────────────

22link    "moduleResolution": "node",  // How to resolve modules

23link    "esModuleInterop": true,     // Better CJS/ESM interop

24link    "allowSyntheticDefaultImports": true,

25link    "resolveJsonModule": true,   // import data from './data.json'

26link    "baseUrl": ".",              // Base for non-relative imports

27link    "paths": {

29link      "@services/*": ["src/services/*"]

30link    },

31link

34link    "types": ["node", "jest"],   // Only include these @types
