linkDecorators

Decorators are a powerful metaprogramming feature that let you annotate and modify classes, methods, properties, and parameters. They're widely used in frameworks like NestJS, TypeORM, and Angular.

Enable decorators in tsconfig.json:

1link{ "compilerOptions": { "experimentalDecorators": true, "emitDecoratorMetadata": true } }

1link// A decorator is a function applied with @

2linkfunction Sealed(constructor: Function) {

3link  Object.seal(constructor);

4link  Object.seal(constructor.prototype);

5link}

7linkfunction Singleton<T extends new (...args: any[]) => {}>(Base: T) {

8link  let instance: InstanceType<T>;

9link  return class extends Base {

10link    constructor(...args: any[]) {

11link      super(...args);

12link      if (!instance) {

13link        instance = this as unknown as InstanceType<T>;

14link      }

15link      return instance;

16link    }

17link  } as T;

18link}

19link

20link@Sealed

21link@Singleton

22linkclass AppConfig {

23link  port = 3000;

24link  db = "mongodb://localhost/app";

25link}

26link

27linkconst a = new AppConfig();

28linkconst b = new AppConfig();

29linkconsole.log(a === b); // true — singleton

linkMethod Decorators

1linkfunction Log(target: any, key: string, descriptor: PropertyDescriptor) {

2link  const original = descriptor.value;

3link

4link  descriptor.value = function (...args: any[]) {

5link    console.log(`▶ ${key}(${args.join(", ")})`);

6link    const result = original.apply(this, args);

7link    console.log(`◀ ${key} → ${result}`);

8link    return result;

9link  };

10link

11link  return descriptor;

12link}

13link

14linkfunction Memoize(target: any, key: string, descriptor: PropertyDescriptor) {

15link  const original = descriptor.value;

16link  const cache = new Map<string, any>();

17link

18link  descriptor.value = function (...args: any[]) {

19link    const cacheKey = JSON.stringify(args);

20link    if (cache.has(cacheKey)) return cache.get(cacheKey);

21link    const result = original.apply(this, args);

22link    cache.set(cacheKey, result);

23link    return result;

24link  };

25link

26link  return descriptor;

27link}

28link

29linkclass MathService {

30link  @Log

31link  add(a: number, b: number): number {

32link    return a + b;

33link  }

34link

35link  @Memoize

36link  fibonacci(n: number): number {

37link    if (n <= 1) return n;

38link    return this.fibonacci(n - 1) + this.fibonacci(n - 2);

39link  }

40link}

linkProperty Decorators

1linkfunction Required(target: any, propertyKey: string) {

2link  let value: any;

3link

4link  Object.defineProperty(target, propertyKey, {

5link    get() { return value; },

6link    set(newValue: any) {

7link      if (newValue === null || newValue === undefined || newValue === "") {

8link        throw new Error(`${propertyKey} is required`);

9link      }

10link      value = newValue;

11link    },

12link  });

13link}

14link

15linkfunction Range(min: number, max: number) {

16link  return function (target: any, propertyKey: string) {

17link    let value: number;

18link

19link    Object.defineProperty(target, propertyKey, {

20link      get() { return value; },

21link      set(newValue: number) {

22link        if (newValue < min || newValue > max) {

23link          throw new RangeError(`${propertyKey} must be between ${min} and ${max}`);

24link        }

25link        value = newValue;

26link      },

27link    });

28link  };

29link}

30link

31linkclass User {

32link  @Required

33link  name!: string;

34link

35link  @Range(0, 150)

36link  age!: number;

37link}

38link

39linkconst user = new User();

40linkuser.name = "Alice"; // OK

41linkuser.age = 200;       // Error: age must be between 0 and 150

linkParameter Decorators

1linkimport "reflect-metadata";

2link

3linkconst REQUIRED_KEY = Symbol("required");

4link

5linkfunction RequiredParam(target: any, methodName: string, paramIndex: number) {

6link  const existingRequired: number[] =

7link    Reflect.getMetadata(REQUIRED_KEY, target, methodName) ?? [];

8link  existingRequired.push(paramIndex);

9link  Reflect.defineMetadata(REQUIRED_KEY, existingRequired, target, methodName);

10link}

11link

12linkfunction ValidateRequired(target: any, methodName: string, descriptor: PropertyDescriptor) {

13link  const original = descriptor.value;

14link

15link  descriptor.value = function (...args: any[]) {

16link    const requiredParams: number[] =

17link      Reflect.getMetadata(REQUIRED_KEY, target, methodName) ?? [];

18link

19link    requiredParams.forEach((index) => {

20link      if (args[index] === undefined || args[index] === null) {

21link        throw new Error(`Parameter ${index} of ${methodName} is required`);

22link      }

23link    });

24link

25link    return original.apply(this, args);

26link  };

27link}

28link

29linkclass UserService {

30link  @ValidateRequired

31link  createUser(@RequiredParam name: string, @RequiredParam email: string) {

32link    return { name, email };

33link  }

34link}

linkDecorator Factories

1link// Decorator factory: returns a decorator

2linkfunction Retry(times: number, delayMs: number = 0) {

3link  return function (target: any, key: string, descriptor: PropertyDescriptor) {

4link    const original = descriptor.value;

5link

6link    descriptor.value = async function (...args: any[]) {

7link      let lastError: Error;

8link      for (let attempt = 1; attempt <= times; attempt++) {

9link        try {

10link          return await original.apply(this, args);

11link        } catch (error) {

12link          lastError = error as Error;

13link          console.warn(`Attempt ${attempt}/${times} failed: ${lastError.message}`);

14link          if (attempt < times && delayMs > 0) {

15link            await new Promise((resolve) => setTimeout(resolve, delayMs));

16link          }

17link        }

18link      }

19link      throw lastError!;

20link    };

21link

22link    return descriptor;

23link  };

24link}

25link

26linkclass ApiClient {

27link  @Retry(3, 1000)

28link  async fetchData(url: string): Promise<unknown> {

29link    const response = await fetch(url);

30link    if (!response.ok) throw new Error(`HTTP ${response.status}`);

31link    return response.json();

32link  }

33link}
