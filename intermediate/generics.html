linkGenerics

Generics let you write reusable, type-safe code that works with any type. They're the key to building robust libraries and utilities.

linkWhy Generics?

1link// ❌ Without generics — loses type info

2linkfunction identity(value: any): any {

3link  return value;

4link}

5linkconst n = identity(42); // type: any — useless!

6link

7link// ✅ With generics — type preserved

8linkfunction identity<T>(value: T): T {

9link  return value;

10link}

11linkconst n = identity(42);       // type: number

12linkconst s = identity("hello");  // type: string

13linkconst b = identity(true);     // type: boolean

linkGeneric Functions

1link// Swap tuple elements

2linkfunction swap<T, U>(pair: [T, U]): [U, T] {

3link  return [pair[1], pair[0]];

4link}

5link

6linkconst result = swap([1, "hello"]); // type: [string, number]

7link

8link// First element of an array

9linkfunction first<T>(arr: T[]): T | undefined {

10link  return arr[0];

11link}

12link

13link// Map with type safety

14linkfunction mapArray<T, U>(arr: T[], transform: (item: T) => U): U[] {

15link  return arr.map(transform);

16link}

17link

18linkconst lengths = mapArray(["hello", "world"], (s) => s.length);

19link// type: number[]

linkGeneric Constraints

1link// T must have a 'length' property

2linkfunction longest<T extends { length: number }>(a: T, b: T): T {

3link  return a.length >= b.length ? a : b;

4link}

5link

6linklongest("hello", "world");      // OK

7linklongest([1, 2, 3], [1, 2]);    // OK

8linklongest({ length: 5 }, { length: 3 }); // OK

9link// longest(1, 2); // Error: number doesn't have 'length'

10link

11link// T must be a key of U

12linkfunction getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {

13link  return obj[key];

14link}

15link

16linkconst user = { id: 1, name: "Alice", email: "a@x.com" };

17linkconst name = getProperty(user, "name"); // type: string

18link// getProperty(user, "missing");        // Error!

linkGeneric Classes

1linkclass Stack<T> {

2link  private items: T[] = [];

3link

4link  push(item: T): void {

5link    this.items.push(item);

6link  }

7link

8link  pop(): T | undefined {

9link    return this.items.pop();

10link  }

11link

12link  peek(): T | undefined {

13link    return this.items[this.items.length - 1];

14link  }

15link

16link  get size(): number {

17link    return this.items.length;

18link  }

19link

20link  isEmpty(): boolean {

21link    return this.items.length === 0;

22link  }

23link}

24link

25linkconst numStack = new Stack<number>();

26linknumStack.push(1);

27linknumStack.push(2);

28linkconsole.log(numStack.pop()); // 2

29link

30linkconst strStack = new Stack<string>();

31linkstrStack.push("hello");

2link  url: string;

3link  method: "GET" | "POST" | "PUT" | "DELETE";

4link  body?: TBody;

5link  params?: TParams;

linkConditional Types

1link// T extends U ? X : Y

2linktype IsString<T> = T extends string ? true : false;

3link

4linktype A = IsString<string>;  // true

5linktype B = IsString<number>;  // false

6linktype C = IsString<"hello">; // true (string literal extends string)

7link

8link// Practical: unwrap arrays

9linktype Flatten<T> = T extends Array<infer U> ? U : T;

10link

11linktype Str = Flatten<string[]>;   // string

12linktype Num = Flatten<number>;     // number (not an array, returns as-is)

13link

14link// Distributive conditional types

15linktype ToArray<T> = T extends any ? T[] : never;

16linktype Result = ToArray<string | number>; // string[] | number[]

linkInfer Keyword

Extract types from other types:

1link// Extract return type of a function

2linktype ReturnType<T extends (...args: any) => any> =

3link  T extends (...args: any) => infer R ? R : never;

4link

5link// Extract first parameter type

6linktype FirstParam<T extends (...args: any) => any> =

7link  T extends (first: infer F, ...rest: any) => any ? F : never;

8link

9link// Unwrap Promise

10linktype Awaited<T> = T extends Promise<infer U> ? Awaited<U> : T;

11link

12link// Usage

13linkasync function fetchUser() {

14link  return { id: 1, name: "Alice" };

15link}

16link

17linktype UserType = Awaited<ReturnType<typeof fetchUser>>;

18link// { id: number; name: string }

linkMapped Types

Transform all properties of a type:

1link// Make all properties optional (like built-in Partial<T>)

2linktype MyPartial<T> = {

3link  [K in keyof T]?: T[K];

4link};

5link

6link// Make all properties required

7linktype MyRequired<T> = {

8link  [K in keyof T]-?: T[K];

9link};

10link

11link// Make all properties readonly

12linktype MyReadonly<T> = {

13link  readonly [K in keyof T]: T[K];

14link};

15link

16link// Transform values

17linktype Stringify<T> = {

18link  [K in keyof T]: string;

19link};

20link

21link// Remap keys

22linktype Getters<T> = {

23link  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];

24link};

25link

26linkinterface User { name: string; age: number; }

27linktype UserGetters = Getters<User>;

28link// { getName: () => string; getAge: () => number }
