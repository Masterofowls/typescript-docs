linkModules & Imports

TypeScript's module system lets you organise code across files with full type safety. Understanding how modules work is critical for large applications.

linkES Modules

1link// math.ts — Named exports

2linkexport function add(a: number, b: number): number {

3link  return a + b;

4link}

5link

6linkexport function subtract(a: number, b: number): number {

7link  return a - b;

8link}

9link

10linkexport const PI = 3.14159;

11link

12link// Default export

13linkexport default class Calculator {

14link  add = add;

15link  subtract = subtract;

16link}

1link// main.ts — Importing

2linkimport Calculator, { add, subtract, PI } from "./math";

3link

4linkconsole.log(add(2, 3));       // 5

5linkconsole.log(PI);               // 3.14159

6linkconst calc = new Calculator();

linkPath Aliases

Configure in tsconfig.json:

1link{

2link  "compilerOptions": {

3link    "baseUrl": ".",

4link    "paths": {

5link      "@/*": ["src/*"],

6link      "@services/*": ["src/services/*"],

7link      "@models/*": ["src/models/*"]

8link    }

9link  }

10link}

1link// Before aliases

2linkimport { UserService } from "../../../services/user.service";

3link

4link// After aliases

5linkimport { UserService } from "@services/user.service";

linkModule Augmentation

Extend an existing module's types:

1link// Express request augmentation

2linkimport "express";

3link

4linkdeclare module "express" {

5link  interface Request {

6link    user?: {

7link      id: string;

8link      email: string;

9link      role: "admin" | "user";

10link    };

11link    requestId?: string;

12link  }

13link}

14link

15link// Now TypeScript knows req.user exists

16linkapp.use((req, res, next) => {

17link  req.user = { id: "1", email: "a@x.com", role: "user" };

18link  next();

19link});

linkAmbient Modules

Declare types for modules without TypeScript support:

1link// @types/my-module/index.d.ts or in a .d.ts file

2linkdeclare module "legacy-lib" {

3link  export function doLegacyThing(input: string): number;

4link  export const version: string;

5link}

6link

7link// wildcard module — for CSS/image imports

8linkdeclare module "*.css" {

9link  const content: Record<string, string>;

10link  export default content;

11link}

12link

13linkdeclare module "*.svg" {

14link  const src: string;

15link  export default src;

16link}
