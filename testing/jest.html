linkTesting a Class

1link// src/services/__tests__/user.service.test.ts

2linkimport { UserService } from "../user.service";

3linkimport { UserRepository } from "../../repositories/user.repository";

4linkimport { AppError } from "../../middleware/error.middleware";

5link

6link// Mock the entire module

7linkjest.mock("../../repositories/user.repository");

8link

9linkconst MockUserRepository = UserRepository as jest.MockedClass<typeof UserRepository>;

10link

11linkdescribe("UserService", () => {

12link  let service: UserService;

13link  let mockRepo: jest.Mocked<UserRepository>;

14link

15link  beforeEach(() => {

16link    // Clear all mocks before each test

17link    jest.clearAllMocks();

18link

19link    // Create a fresh mock instance

20link    mockRepo = new MockUserRepository() as jest.Mocked<UserRepository>;

21link    service = new UserService(mockRepo);

22link  });

23link

24link  describe("findById", () => {

25link    it("returns sanitized user when found", async () => {

26link      const rawUser = {

27link        id: "1",

28link        name: "Alice",

29link        email: "alice@example.com",

30link        password: "hashed",

31link        role: "user" as const,

32link        createdAt: new Date(),

33link        updatedAt: new Date(),

34link      };

35link

36link      mockRepo.findOne.mockResolvedValueOnce(rawUser);

37link

38link      const user = await service.findById("1");

39link

40link      expect(user).not.toBeNull();

41link      expect(user!.name).toBe("Alice");

42link      expect(user).not.toHaveProperty("password"); // sanitized

43link      expect(mockRepo.findOne).toHaveBeenCalledWith({ where: { id: "1" } });

44link    });

45link

46link    it("returns null when user not found", async () => {

47link      mockRepo.findOne.mockResolvedValueOnce(null);

48link      const user = await service.findById("999");

49link      expect(user).toBeNull();

50link    });

51link  });

52link

53link  describe("create", () => {

54link    it("throws 409 if email already exists", async () => {

55link      mockRepo.findOne.mockResolvedValueOnce({ id: "1", email: "a@b.com" } as any);

56link

57link      await expect(

58link        service.create({ name: "Bob", email: "a@b.com", password: "pass123!" })

59link      ).rejects.toThrow(AppError);

60link

61link      await expect(

62link        service.create({ name: "Bob", email: "a@b.com", password: "pass123!" })

63link      ).rejects.toMatchObject({ statusCode: 409 });

64link    });

65link

66link    it("hashes the password before saving", async () => {

67link      mockRepo.findOne.mockResolvedValueOnce(null);

68link      mockRepo.create.mockResolvedValueOnce({

69link        id: "2",

70link        name: "Carol",

71link        email: "carol@example.com",

72link        password: "$2b$...",

73link        role: "user",

74link        createdAt: new Date(),

75link        updatedAt: new Date(),

76link      });

77link

78link      await service.create({ name: "Carol", email: "carol@example.com", password: "mypassword" });

79link

80link      const savedData = mockRepo.create.mock.calls[0][0];

81link      expect(savedData.password).not.toBe("mypassword");

82link      expect(savedData.password).toMatch(/^\$2[aby]\$/); // bcrypt hash

83link    });

84link  });

85link});

linkCustom Matchers

1link// src/test/matchers.ts

2linkexpect.extend({

3link  toBeValidEmail(received: string) {

4link    const pass = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(received);

5link    return {

6link      pass,

7link      message: () => pass

8link        ? `Expected ${received} NOT to be a valid email`

9link        : `Expected ${received} to be a valid email`,

10link    };

11link  },

12link

13link  toHaveStatusCode(received: Response, expected: number) {

14link    const pass = received.status === expected;

15link    return {

16link      pass,

17link      message: () => `Expected status ${expected}, got ${received.status}`,

18link    };

19link  },

20link});

21link

22link// Type augmentation for custom matchers

23linkdeclare global {

24link  namespace jest {

25link    interface Matchers<R> {

26link      toBeValidEmail(): R;

27link      toHaveStatusCode(code: number): R;

28link    }

29link  }

30link}
